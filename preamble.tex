% Helper macro for endpoint circles
\newcommand{\numberlineendpoint}[2]{%
  \ifthenelse{\equal{#2}{closed}}{%
    \filldraw[black] (#1,0.3) circle (3pt);
  }{%
    \draw[black, thick] (#1,0.3) circle (3pt);
  }%
}

% Main macro for number line with interval and endpoint styles
\newcommand{\numberlineinterval}[6][0]{%
  \begin{tikzpicture}[baseline=-0.5ex]
    \draw[latex-latex] (#1,0) -- (#2,0);
    \foreach \x in {#1,...,#2} {
      \draw (\x,0.1) -- (\x,-0.1) node[anchor=north] {\x};
    }
    \draw[thick,black] (#3,0.3) -- (#4,0.3);
    \numberlineendpoint{#3}{#5}
    \numberlineendpoint{#4}{#6}
  \end{tikzpicture}%
}
